    const menuBtn = document.getElementById('menuBtn');
    const menu = document.getElementById('dropdownMenu');
    let isOpen = false;

    gsap.set(menu, { y: -50, opacity: 0, display: "none" });

    menuBtn.addEventListener('click', () => {
      if (!isOpen) {
        gsap.to(menu, {
          y: 0,
          opacity: 1,
          duration: 0.8,
          ease: "power4.out",
          display: "block"
        });
      } else {
        gsap.to(menu, {
          y: -50,
          opacity: 0,
          duration: 0.5,
          ease: "power4.in",
          onComplete: () => (menu.style.display = "none")
        });
      }
      isOpen = !isOpen;
      menuBtn.classList.toggle("active");
    });
  </script>
